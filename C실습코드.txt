여행 총 경비 구하기
--------------------------------------------------------------
#include<stdio.h>
int main(void)
{

	int sum;    //총액을 저장하는 변수
	int price;  // 각 항목의 가격을 저장하는 변수
	int days;   // 몇 박인지 저장하는 변수

	sum = 0;

	printf("여행은 몇박인가요? :");
	scanf_s("%d", &days);

	printf("항공권 가격: ");
	scanf_s("%d", &price);
	sum = sum + price;

	printf("호텔 1박 가격:");
	scanf_s(" %d", &price);
	sum = sum + days * price;

	printf("하루에 필요한 용돈:");
	scanf_s("%d", &price);
	sum = sum + days * price;

	printf("===============================\n");
	printf(" 총 여행 비용 : %d \n", sum);
	printf("===============================\n");
	return 0;


}

초를 받아 시간 분 초로 나타내기
#include<stdio.h>
#define SEC_PER_MINUTE 60 // 1분은 60초
#define SEC_PER_HOUR 3600 // 1시간은 3600초

int main(void)
{
	int input, hour, res, minute, second;

	printf("초단위의 시간을 입력하시오:(21억초이하)");
	scanf_s("%d", &input); //초단위의 시간을 읽는다.

	hour = input / SEC_PER_HOUR;
	res = input % SEC_PER_HOUR;

	minute = res / SEC_PER_MINUTE;
	second = res % SEC_PER_MINUTE;

	printf("%d초는 %d시간 %d분 %d초 입니다.\n", input, hour, minute, second);

	return 0;

}
